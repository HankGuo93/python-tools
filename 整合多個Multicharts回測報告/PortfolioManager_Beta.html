def GetTradeData(fileName):
    TradeData = pd.read_excel("./" + fileName, 
                        sheet_name='',
                        skiprows = range(0, 2),
                        usecols = "E:J")
    TradeData_df = pd.DataFrame(TradeData, columns=['','()'])
    TradeData_df['()'].fillna(method='ffill', inplace=True)
    TradeData_df_filterTheClosing = TradeData_df[np.arange(len(TradeData_df)) % 2 == 1]
    return TradeData_df_filterTheClosing

filesInFolder = []
for subdir, dirs, files in os.walk('./'):
    for file in files:
        filesInFolder.append(file)

reports = list(filter(lambda k: '' in k, filesInFolder))

df_summary = pd.DataFrame({'':datetime(1911, 1, 1),'()':[200100.0]})

for report in reports:
    df_summary = pd.concat([df_summary, GetTradeData(report)], axis=0, sort=True).sort_values(['']).reset_index(drop=True)

tradeFee = 50
    
df_summary['()'] = df_summary['()'] - tradeFee*2
df_summary['TotalEquity'] = df_summary['()'].cumsum()
